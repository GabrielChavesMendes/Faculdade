F_BCD_TO_HEX: MOV A, dado01
ANI 0FH
MOV dado02, A
MOV A, dado01
ANI F0H
RRC
RRC
RRC
RRC
MOV dado01, A
MVI A 0AH
CALL MULTIPLY
ADD dado02
RET

MULTIPLY: MVI dado02, 00H
MULT_LOOP: CMP D
JZ END_MULTIPLY
ADD dado02
DCR D
JMP MULT_LOOP
END_MULT: MOV A, dado02
RET 

MAIN: LXI H, 0050G
MOV dado01, M
CALL F_BCD_TO_HEX
STA 0051H
HLT
